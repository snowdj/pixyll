# sid <- "orcl12c"
service <- "xe"
connect.string <- paste(
"(DESCRIPTION=",
"(ADDRESS=(PROTOCOL=tcp)(HOST=", host, ")(PORT=", port, "))",
"(CONNECT_DATA=(SERVICE_NAME=", service, ")))", sep = "")
 # username = "system", password = "orcl"

dbReadTable(con, "EMP")
